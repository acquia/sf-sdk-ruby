def get_cron_jobs(page = nil, limit = nil)
  args = {}
  args['page'] = page unless page.nil?
  args['limit'] = limit unless limit.nil?
  url = "/api/v1/cronjobs?#{args.map { |k, v| "#{k}=#{v}" }.join('&')}"
  @conn.get(url)
end
