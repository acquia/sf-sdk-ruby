def group_list
  page = 1
  not_done = true
  count = 0
  while not_done
    current_path = '/api/v1/groups?page=' << page.to_s
    res = @conn.get(current_path)
    if res['groups'] == []
      not_done = false
    elsif !res['message'].nil?
      return { 'message' => res['message'] }
    elsif page == 1
      count = res['count']
      groups = res['groups']
    else
      res['groups'].each do |group|
        groups << group
      end
    end
    page += 1
  end
  { 'count' => count, 'groups' => groups }
end
